    70→function GenerationStepProgress() {
    71→  const { locale } = useI18n();
    72→  const generationProgress = useZDesignStore((s) => s.generationProgress);
    73→  const creativeMode = useZDesignStore((s) => s.creativeMode);
    74→  const isDe = locale === 'de';
    75→
    76→  const steps = [
    77→    { id: 'analyzing', label: isDe ? 'Analysiere...' : 'Analyzing request...', icon: Brain },
    78→    { id: 'generating', label: isDe ? 'Generiere Layout...' : 'Generating layout...', icon: Sparkles },
    79→    { id: 'rendering', label: isDe ? 'Füge Stile hinzu...' : 'Adding styles...', icon: Palette },
    80→    { id: 'evaluating', label: isDe ? 'Finalisiere...' : 'Finalizing design...', icon: CheckCircle2 },
    81→  ];
    82→
    83→  const currentIdx = steps.findIndex((s) => s.id === generationProgress.stage);
    84→
    85→  return (
    86→    <div className="px-4 py-2 space-y-2">
    87→      {/* Main status line */}
    88→      <div className="flex items-center gap-2">
    89→        <Loader2 className="size-3.5 animate-spin text-emerald-500 shrink-0" />
    90→        <span className="text-xs font-medium text-foreground">
    91→          {isDe ? 'KI generiert Design...' : 'Generating design with AI...'}
    92→        </span>
    93→        {generationProgress.estimatedTimeLeft && generationProgress.estimatedTimeLeft > 0 && (
    94→          <span className="text-[10px] text-muted-foreground ml-auto">
    95→            ~{generationProgress.estimatedTimeLeft}s {isDe ? 'übrig' : 'remaining'}
    96→          </span>
    97→        )}
    98→      </div>
    99→
   100→      {/* Progress bar */}
   101→      <Progress value={generationProgress.percentage} className="h-1.5 [&>div]:bg-emerald-500" />
   102→
   103→      {/* Steps */}
   104→      <div className="flex items-center gap-1">
   105→        {steps.map((step, idx) => {
   106→          const StepIcon = step.icon;
   107→          const isActive = step.id === generationProgress.stage;
   108→          const isDone = idx < currentIdx || generationProgress.stage === 'complete';
   109→
   110→          return (
   111→            <div key={step.id} className="flex items-center gap-1 flex-1 min-w-0">
   112→              <div className={`flex items-center gap-1 px-1.5 py-0.5 rounded text-[10px] font-medium truncate ${
   113→                isActive
   114→                  ? 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400'
   115→                  : isDone
   116→                    ? 'bg-emerald-50 text-emerald-600 dark:bg-emerald-900/20 dark:text-emerald-500'
   117→                    : 'bg-muted/50 text-muted-foreground'
   118→              }`}>
   119→                {isActive ? (
   120→                  <Loader2 className="size-2.5 animate-spin shrink-0" />
   121→                ) : isDone ? (
   122→                  <CheckCircle2 className="size-2.5 shrink-0" />
   123→                ) : (
   124→                  <StepIcon className="size-2.5 shrink-0" />
   125→                )}
   126→                <span className="truncate">{step.label}</span>
   127→              </div>
   128→              {idx < steps.length - 1 && (
   129→                <div className={`h-px flex-1 min-w-[8px] ${isDone ? 'bg-emerald-300 dark:bg-emerald-700' : 'bg-border'}`} />
   130→              )}
   131→            </div>
   132→          );
   133→        })}
   134→      </div>
   135→
   136→      {/* Creative mode indicator */}
   137→      {creativeMode && (
   138→        <div className="flex items-center gap-1 text-[10px] text-amber-600 dark:text-amber-400">
   139→          <Zap className="size-2.5" />
   140→          <span className="font-medium">Creative Mode active</span>
   141→        </div>
   142→      )}
   143→
   144→      {/* Status message */}
   145→      {generationProgress.message && (
   146→        <p className="text-[10px] text-muted-foreground">{generationProgress.message}</p>

   184→  onEnhance: () => void;

   239→            </div>
   240→          )}
   241→        </div>
   242→        {/* Quality Score Badge for design messages */}
   243→        {showQuality && (
   244→          <motion.div
   245→            initial={{ opacity: 0, y: 4 }}
   246→            animate={{ opacity: 1, y: 0 }}
   247→            transition={{ delay: 0.2 }}

   249→          >
   250→            <div className={`flex items-center gap-1.5 px-2.5 py-1 rounded-lg text-xs font-medium ${

   252→                ? 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400'
   253→                : qualityReport.overallScore >= 60
   254→                  ? 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400'

   256→            }`}>
   257→              <CheckCircle2 className="size-3" />
   258→              Quality: {qualityReport.overallScore}/100
   259→              <div className="flex gap-0.5 ml-1">
   260→                {[
   261→                  { label: 'CSS', value: qualityReport.cssValid },
   262→                  { label: 'Sem', value: qualityReport.semantics },
   263→                  { label: 'Rsp', value: qualityReport.responsiveness },

   277→                size="sm"

   279→                className="h-7 text-[11px] px-2.5 gap-1 border-amber-300 text-amber-700 hover:bg-amber-50 dark:border-amber-800 dark:text-amber-400 dark:hover:bg-amber-900/20"

   290→            )}
   291→          </motion.div>
   292→        )}
   293→        {mounted && (
   294→          <p
   295→            className={`text-[10px] text-muted-foreground px-1 ${isUser ? 'text-right' : ''}`}
   296→          >
   297→            {new Date(message.createdAt).toLocaleTimeString([], {
   298→              hour: '2-digit',
   299→              minute: '2-digit',
   300→            })}
   301→          </p>
   302→        )}
   303→      </div>
   304→    </motion.div>
   305→  );
   306→}

   366→  const chatMessages = useZDesignStore((s) => s.chatMessages);
   367→  const addChatMessage = useZDesignStore((s) => s.addChatMessage);
   368→  const clearChat = useZDesignStore((s) => s.clearChat);
   369→  const isGenerating = useZDesignStore((s) => s.isGenerating);
   370→  const setIsGenerating = useZDesignStore((s) => s.setIsGenerating);
   371→  const projectId = useZDesignStore((s) => s.projectId);
   372→  const designTree = useZDesignStore((s) => s.designTree);
   373→  const designSystem = useZDesignStore((s) => s.designSystem);
   374→  const setDesignTree = useZDesignStore((s) => s.setDesignTree);
   375→  const setGenerationProgress = useZDesignStore((s) => s.setGenerationProgress);
   376→  const resetGenerationProgress = useZDesignStore((s) => s.resetGenerationProgress);
   377→  const setQualityReport = useZDesignStore((s) => s.setQualityReport);
   378→  const qualityReport = useZDesignStore((s) => s.qualityReport);
   379→  const generationProgress = useZDesignStore((s) => s.generationProgress);
   380→
   381→  const [input, setInput] = useState('');
   382→  const [isEnhancing, setIsEnhancing] = useState(false);
   383→  const scrollRef = useRef<HTMLDivElement>(null);
   384→  const textareaRef = useRef<HTMLTextAreaElement>(null);
   385→
   386→  // AI Image dialog
   387→  const [aiImageOpen, setAiImageOpen] = useState(false);
   388→
   389→  // Research dialog
   390→  const [researchOpen, setResearchOpen] = useState(false);
   391→
   392→  // === Refs for mutable state to avoid stale closures ===
   393→  const projectIdRef = useRef(projectId);
   394→  const isGeneratingRef = useRef(isGenerating);
   395→  const chatMessagesRef = useRef(chatMessages);
   396→  const designTreeRef = useRef(designTree);
   397→  const designSystemRef = useRef(designSystem);
   398→
   399→  useEffect(() => { projectIdRef.current = projectId; }, [projectId]);
   400→  useEffect(() => { isGeneratingRef.current = isGenerating; }, [isGenerating]);
   401→  useEffect(() => { chatMessagesRef.current = chatMessages; }, [chatMessages]);
   402→  useEffect(() => { designTreeRef.current = designTree; }, [designTree]);
   403→  useEffect(() => { designSystemRef.current = designSystem; }, [designSystem]);
   404→
   405→  // Stable store action refs
   406→  const addChatMessageRef = useRef(addChatMessage);
   407→  const setIsGeneratingRef = useRef(setIsGenerating);
   408→  const setDesignTreeRef = useRef(setDesignTree);
   409→  const setGenerationProgressRef = useRef(setGenerationProgress);
   410→  const resetGenerationProgressRef = useRef(resetGenerationProgress);
   411→  const setQualityReportRef = useRef(setQualityReport);
   412→  const tRef = useRef(t);
   413→
   414→  useEffect(() => { addChatMessageRef.current = addChatMessage; }, [addChatMessage]);
   415→  useEffect(() => { setIsGeneratingRef.current = setIsGenerating; }, [setIsGenerating]);
   416→  useEffect(() => { setDesignTreeRef.current = setDesignTree; }, [setDesignTree]);
   417→  useEffect(() => { setGenerationProgressRef.current = setGenerationProgress; }, [setGenerationProgress]);
   418→  useEffect(() => { resetGenerationProgressRef.current = resetGenerationProgress; }, [resetGenerationProgress]);
   419→  useEffect(() => { setQualityReportRef.current = setQualityReport; }, [setQualityReport]);
   420→  useEffect(() => { tRef.current = t; }, [t]);
   421→
   422→  // ============ Progress simulation ============
   423→  // Simulates step-by-step progress while the API call is in flight
   424→  // LLM takes ~60-90s, so we simulate progress over that duration
   425→  const progressTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);
   426→
   427→  const startProgressSimulation = useCallback(() => {
   428→    // Only set up incremental percentage updates during the API call
   429→    // Initial stage is already set by sendMessage
   430→    const startTime = Date.now();
   431→    const totalDuration = 80000; // 80 seconds estimated total for LLM generation
   432→
   433→    // Auto-advance stages on a timer
   434→    // analyzing: 0-5s, generating: 5-55s, rendering: handled after API returns
   435→    const stageTimers: Array<{ delay: number; stage: string; pct: number; msg: string }> = [
   436→      { delay: 5000, stage: 'generating', pct: 25, msg: 'AI is creating your design...' },
   437→      { delay: 20000, stage: 'generating', pct: 35, msg: 'Building layout structure...' },
   438→      { delay: 40000, stage: 'generating', pct: 50, msg: 'Adding components and styles...' },
   439→      { delay: 60000, stage: 'generating', pct: 60, msg: 'Polishing design details...' },
   440→    ];
   441→
   442→    const stageTimeouts: Array<ReturnType<typeof setTimeout>> = [];
   443→    for (const st of stageTimers) {
   444→      stageTimeouts.push(setTimeout(() => {
   445→        setGenerationProgressRef.current({
   446→          stage: st.stage as 'analyzing' | 'generating' | 'rendering' | 'evaluating' | 'complete' | 'error' | 'idle',
   447→          stageLabel: st.msg,
   448→          percentage: st.pct,
   449→          message: st.msg,
   450→          estimatedTimeLeft: Math.max(0, Math.round((totalDuration - st.delay) / 1000)),
   451→        });
   452→      }, st.delay));
   453→    }
   454→
   455→    progressTimerRef.current = setInterval(() => {
   456→      const elapsed = Date.now() - startTime;
   457→      const pct = Math.min(65, Math.round(20 + (elapsed / totalDuration) * 45)); // 20-65% range
   458→      const timeLeft = Math.max(0, Math.round((totalDuration - elapsed) / 1000));
   459→
   460→      // Update percentage and time remaining
   461→      setGenerationProgressRef.current({
   462→        percentage: pct,
   463→        estimatedTimeLeft: timeLeft,
   464→      });
   465→    }, 1000);
   466→
   467→    // Store stage timeouts for cleanup
   468→    (progressTimerRef as React.MutableRefObject<ReturnType<typeof setInterval> | null & { stageTimeouts?: Array<ReturnType<typeof setTimeout>> }>).current = {
   469→      ...progressTimerRef.current,
   470→      stageTimeouts,
   471→    } as unknown as ReturnType<typeof setInterval>;
   472→  }, []);
   473→
   474→  const stopProgressSimulation = useCallback((success: boolean) => {
   475→    if (progressTimerRef.current) {
   476→      clearInterval(progressTimerRef.current);
   477→      // Clean up stage timeouts
   478→      const ref = progressTimerRef.current as unknown as { stageTimeouts?: Array<ReturnType<typeof setTimeout>> };
   479→      if (ref.stageTimeouts) {
   480→        ref.stageTimeouts.forEach(clearTimeout);
   481→      }
   482→      progressTimerRef.current = null;
   483→    }
   484→
   485→    if (success) {
   486→      // If we're still in evaluating stage from sendMessage, set complete
   487→      // If we haven't reached evaluating yet, skip to complete
   488→      setGenerationProgressRef.current({
   489→        stage: 'complete',
   490→        stageLabel: 'Complete!',
   491→        percentage: 100,
   492→        message: 'Design generated successfully!',
   493→      });
   494→
   495→      // Reset after a short delay
   496→      setTimeout(() => {
   497→        resetGenerationProgressRef.current();
   498→      }, 1500);
   499→    } else {
   500→      setGenerationProgressRef.current({
   501→        stage: 'error',
   502→        stageLabel: 'Error',
   503→        percentage: 100,
   504→        message: 'Generation failed',
   505→      });
   506→
   507→      setTimeout(() => {
   508→        resetGenerationProgressRef.current();
   509→      }, 3000);
   510→    }
   511→  }, []);
   512→
   513→  // ============ Evaluate Design Quality ============
   514→  const evaluateDesignQuality = useCallback((designTree: Record<string, unknown>) => {
   515→    const report = {
   516→      overallScore: 0,
   517→      completeness: 0,
   518→      cssValid: 0,
   519→      semantics: 0,
   520→      responsiveness: 0,
   521→      accessibility: 0,
   522→      issues: [] as Array<{
   523→        severity: 'critical' | 'warning' | 'info';
   524→        category: 'css' | 'semantics' | 'responsiveness' | 'accessibility' | 'completeness';
   525→        message: string;
   526→        autoFixable: boolean;
   527→      }>,
   528→      suggestions: [] as string[],
   529→    };
   530→
   531→    // Check completeness
   532→    const children = (designTree as { children?: unknown[] }).children;
   533→    if (children && children.length > 0) {
   534→      report.completeness = Math.min(100, children.length * 20);
   535→      // Check for essential sections
   536→      const hasNav = children.some((c: unknown) => {
   537→        const node = c as { type?: string; tag?: string };
   538→        return node.type === 'nav' || node.tag === 'nav' || node.type === 'header';
   539→      });
   540→      const hasFooter = children.some((c: unknown) => {
   541→        const node = c as { type?: string; tag?: string };
   542→        return node.type === 'footer' || node.tag === 'footer';
   543→      });
   544→      const hasContent = children.some((c: unknown) => {
   545→        const node = c as { type?: string; tag?: string };
   546→        return node.type === 'section' || node.tag === 'section' || node.type === 'main';
   547→      });
   548→
   549→      if (!hasNav) {
   550→        report.issues.push({
   551→          severity: 'warning',
   552→          category: 'completeness',
   553→          message: 'No navigation section found',
   554→          autoFixable: false,
   555→        });
   556→      }
   557→      if (!hasFooter) {
   558→        report.issues.push({
   559→          severity: 'info',
   560→          category: 'completeness',
   561→          message: 'No footer section found',
   562→          autoFixable: false,
   563→        });
   564→      }
   565→      if (!hasContent) {
   566→        report.issues.push({
   567→          severity: 'warning',
   568→          category: 'completeness',
   569→          message: 'No main content section found',
   570→          autoFixable: false,
   571→        });
   572→      }
   573→
   574→      if (hasNav && hasFooter && hasContent) {
   575→        report.completeness = 100;
   576→      }
   577→    } else {
   578→      report.issues.push({
   579→        severity: 'critical',
   580→        category: 'completeness',
   581→        message: 'Design tree has no children',
   582→        autoFixable: false,
   583→      });
   584→    }
   585→
   586→    // Check CSS validity (basic check)
   587→    const styleStr = JSON.stringify(designTree);
   588→    const hasTailwindShorthand = /\d+xl|text-md|p-lg|gap-md/.test(styleStr);
   589→    if (hasTailwindShorthand) {
   590→      report.cssValid = 50;
   591→      report.issues.push({
   592→        severity: 'critical',
   593→        category: 'css',
   594→        message: 'Found Tailwind shorthand in CSS values — should use real CSS (px, rem, %)',
   595→        autoFixable: true,
   596→      });
   597→    } else {
   598→      report.cssValid = 95;
   599→    }
   600→
   601→    // Check semantics
   602→    const semanticTags = ['header', 'nav', 'main', 'section', 'footer', 'article', 'aside'];
   603→    const usedTags = new Set<string>();
   604→    const findTags = (node: unknown) => {
   605→      const n = node as { tag?: string; children?: unknown[] };
   606→      if (n.tag) usedTags.add(n.tag);
   607→      if (n.children) n.children.forEach(findTags);
   608→    };
   609→    findTags(designTree);
   610→    const semanticCount = semanticTags.filter((tag) => usedTags.has(tag)).length;
   611→    report.semantics = Math.min(100, Math.round((semanticCount / 3) * 100)); // 3+ semantic tags = good
   612→
   613→    // Check responsiveness
   614→    const hasFlexOrGrid = /flex|grid/.test(styleStr);
   615→    const hasPercentOrRelative = /%|rem|vh|vw/.test(styleStr);
   616→    report.responsiveness = hasFlexOrGrid ? (hasPercentOrRelative ? 90 : 70) : 40;
   617→
   618→    // Check accessibility basics
   619→    const hasHeadings = /"heading"|"h1"|"h2"|"h3"/.test(styleStr);
   620→    const hasButtons = /"button"/.test(styleStr);
   621→    report.accessibility = hasHeadings ? (hasButtons ? 80 : 60) : 40;
   622→    if (!hasHeadings) {
   623→      report.issues.push({
   624→        severity: 'warning',
   625→        category: 'accessibility',
   626→        message: 'No heading elements found — important for screen readers',
   627→        autoFixable: false,
   628→      });
   629→    }
   630→
   631→    // Suggestions
   632→    if (report.responsiveness < 80) {
   633→      report.suggestions.push('Consider using flexbox/grid layouts with relative units for better responsiveness');
   634→    }
   635→    if (report.semantics < 80) {
   636→      report.suggestions.push('Use semantic HTML tags (header, nav, main, section, footer) for better structure');
   637→    }
   638→    if (report.accessibility < 80) {
   639→      report.suggestions.push('Add proper heading hierarchy and ARIA labels for accessibility');
   640→    }
   641→
   642→    // Overall score
   643→    report.overallScore = Math.round(
   644→      (report.completeness * 0.25 +
   645→        report.cssValid * 0.2 +
   646→        report.semantics * 0.2 +
   647→        report.responsiveness * 0.2 +
   648→        report.accessibility * 0.15)
   649→    );
   650→
   651→    return report;
   652→  }, []);
