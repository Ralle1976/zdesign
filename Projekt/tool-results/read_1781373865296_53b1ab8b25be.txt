    75→  const creativeMode = useZDesignStore((s) => s.creativeMode);

    81→    { id: 'rendering', label: isDe ? 'Füge Stile hinzu...' : 'Adding styles...', icon: Palette },
    82→    { id: 'evaluating', label: isDe ? 'Finalisiere...' : 'Finalizing design...', icon: CheckCircle2 },

    88→    <div className="px-4 py-2 space-y-2">
    89→      {/* Main status line */}

    91→        <Loader2 className="size-3.5 animate-spin text-emerald-500 shrink-0" />

    93→          {isDe ? 'KI generiert Design...' : 'Generating design with AI...'}

    95→        {generationProgress.estimatedTimeLeft && generationProgress.estimatedTimeLeft > 0 && (
    96→          <span className="text-[10px] text-muted-foreground ml-auto">
    97→            ~{generationProgress.estimatedTimeLeft}s {isDe ? 'übrig' : 'remaining'}
    98→          </span>
    99→        )}
   100→      </div>
   101→
   102→      {/* Progress bar */}
   103→      <Progress value={generationProgress.percentage} className="h-1.5 [&>div]:bg-emerald-500" />

   106→      <div className="flex items-center gap-1">

   421→  const chatMessagesRef = useRef(chatMessages);
   422→  const designTreeRef = useRef(designTree);
   423→  const designSystemRef = useRef(designSystem);
   424→
   425→  useEffect(() => { projectIdRef.current = projectId; }, [projectId]);
   426→  useEffect(() => { isGeneratingRef.current = isGenerating; }, [isGenerating]);
   427→  useEffect(() => { chatMessagesRef.current = chatMessages; }, [chatMessages]);

   432→  const addChatMessageRef = useRef(addChatMessage);
   433→  const setIsGeneratingRef = useRef(setIsGenerating);
   434→  const setDesignTreeRef = useRef(setDesignTree);
   435→  const setGenerationProgressRef = useRef(setGenerationProgress);

   439→

   441→  useEffect(() => { setIsGeneratingRef.current = setIsGenerating; }, [setIsGenerating]);
   442→  useEffect(() => { setDesignTreeRef.current = setDesignTree; }, [setDesignTree]);
   443→  useEffect(() => { setGenerationProgressRef.current = setGenerationProgress; }, [setGenerationProgress]);
   444→  useEffect(() => { resetGenerationProgressRef.current = resetGenerationProgress; }, [resetGenerationProgress]);
   445→  useEffect(() => { setQualityReportRef.current = setQualityReport; }, [setQualityReport]);
   446→  useEffect(() => { tRef.current = t; }, [t]);
   447→
   448→  // ============ Progress simulation ============
   449→  // Simulates step-by-step progress while the API call is in flight
   450→  // LLM takes ~60-90s, so we simulate progress over that duration
   451→  const progressTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);
   452→
   453→  const startProgressSimulation = useCallback(() => {
   454→    // Only set up incremental percentage updates during the API call
   455→    // Initial stage is already set by sendMessage
   456→    const startTime = Date.now();
   457→    const totalDuration = 80000; // 80 seconds estimated total for LLM generation
   458→
   459→    // Auto-advance stages on a timer
   460→    // analyzing: 0-5s, generating: 5-55s, rendering: handled after API returns
   461→    const stageTimers: Array<{ delay: number; stage: string; pct: number; msg: string }> = [
   462→      { delay: 5000, stage: 'generating', pct: 25, msg: 'AI is creating your design...' },
   463→      { delay: 20000, stage: 'generating', pct: 35, msg: 'Building layout structure...' },
   464→      { delay: 40000, stage: 'generating', pct: 50, msg: 'Adding components and styles...' },
   465→      { delay: 60000, stage: 'generating', pct: 60, msg: 'Polishing design details...' },
   466→    ];
   467→
   468→    const stageTimeouts: Array<ReturnType<typeof setTimeout>> = [];
   469→    for (const st of stageTimers) {
   470→      stageTimeouts.push(setTimeout(() => {
   471→        setGenerationProgressRef.current({
   472→          stage: st.stage as 'analyzing' | 'generating' | 'rendering' | 'evaluating' | 'complete' | 'error' | 'idle',
   473→          stageLabel: st.msg,
   474→          percentage: st.pct,
   475→          message: st.msg,
   476→          estimatedTimeLeft: Math.max(0, Math.round((totalDuration - st.delay) / 1000)),
   477→        });
   478→      }, st.delay));
   479→    }
   480→
   481→    progressTimerRef.current = setInterval(() => {
   482→      const elapsed = Date.now() - startTime;
   483→      const pct = Math.min(65, Math.round(20 + (elapsed / totalDuration) * 45)); // 20-65% range
   484→      const timeLeft = Math.max(0, Math.round((totalDuration - elapsed) / 1000));
   485→
   486→      // Update percentage and time remaining
   487→      setGenerationProgressRef.current({
   488→        percentage: pct,
   489→        estimatedTimeLeft: timeLeft,
   490→      });
   491→    }, 1000);
   492→
   493→    // Store stage timeouts for cleanup
   494→    (progressTimerRef as React.MutableRefObject<ReturnType<typeof setInterval> | null & { stageTimeouts?: Array<ReturnType<typeof setTimeout>> }>).current = {
   495→      ...progressTimerRef.current,
   496→      stageTimeouts,
   497→    } as unknown as ReturnType<typeof setInterval>;
   498→  }, []);
   499→
   500→  const stopProgressSimulation = useCallback((success: boolean) => {
   501→    if (progressTimerRef.current) {

   539→  // ============ Evaluate Design Quality ============
   540→  const evaluateDesignQuality = useCallback((designTree: Record<string, unknown>) => {
   541→    const report = {
   542→      overallScore: 0,
   543→      completeness: 0,
   544→      cssValid: 0,
   545→      semantics: 0,
   546→      responsiveness: 0,
   547→      accessibility: 0,
   548→      issues: [] as Array<{
   549→        severity: 'critical' | 'warning' | 'info';
   550→        category: 'css' | 'semantics' | 'responsiveness' | 'accessibility' | 'completeness';
   551→        message: string;
   552→        autoFixable: boolean;
   553→      }>,
   554→      suggestions: [] as string[],
   555→    };
   556→
   557→    // Check completeness
   558→    const children = (designTree as { children?: unknown[] }).children;
   559→    if (children && children.length > 0) {
   560→      report.completeness = Math.min(100, children.length * 20);
   561→      // Check for essential sections
   562→      const hasNav = children.some((c: unknown) => {
   563→        const node = c as { type?: string; tag?: string };
   564→        return node.type === 'nav' || node.tag === 'nav' || node.type === 'header';
   565→      });
   566→      const hasFooter = children.some((c: unknown) => {
   567→        const node = c as { type?: string; tag?: string };
   568→        return node.type === 'footer' || node.tag === 'footer';
   569→      });
   570→      const hasContent = children.some((c: unknown) => {
   571→        const node = c as { type?: string; tag?: string };
   572→        return node.type === 'section' || node.tag === 'section' || node.type === 'main';
   573→      });
   574→
   575→      if (!hasNav) {
   576→        report.issues.push({
   577→          severity: 'warning',
   578→          category: 'completeness',
   579→          message: 'No navigation section found',
   580→          autoFixable: false,
   581→        });
   582→      }
   583→      if (!hasFooter) {
   584→        report.issues.push({
   585→          severity: 'info',
   586→          category: 'completeness',
   587→          message: 'No footer section found',
   588→          autoFixable: false,
   589→        });
   590→      }
   591→      if (!hasContent) {
   592→        report.issues.push({
   593→          severity: 'warning',
   594→          category: 'completeness',
   595→          message: 'No main content section found',
   596→          autoFixable: false,
   597→        });
   598→      }
   599→
   600→      if (hasNav && hasFooter && hasContent) {
   601→        report.completeness = 100;
   602→      }
   603→    } else {
   604→      report.issues.push({
   605→        severity: 'critical',
   606→        category: 'completeness',
   607→        message: 'Design tree has no children',
   608→        autoFixable: false,
   609→      });
   610→    }
   611→
   612→    // Check CSS validity (basic check)
   613→    const styleStr = JSON.stringify(designTree);
   614→    const hasTailwindShorthand = /\d+xl|text-md|p-lg|gap-md/.test(styleStr);
   615→    if (hasTailwindShorthand) {
   616→      report.cssValid = 50;
   617→      report.issues.push({
   618→        severity: 'critical',
   619→        category: 'css',
   620→        message: 'Found Tailwind shorthand in CSS values — should use real CSS (px, rem, %)',
   621→        autoFixable: true,
   622→      });
   623→    } else {
   624→      report.cssValid = 95;
   625→    }
   626→
   627→    // Check semantics
   628→    const semanticTags = ['header', 'nav', 'main', 'section', 'footer', 'article', 'aside'];
   629→    const usedTags = new Set<string>();
   630→    const findTags = (node: unknown) => {
   631→      const n = node as { tag?: string; children?: unknown[] };
   632→      if (n.tag) usedTags.add(n.tag);
   633→      if (n.children) n.children.forEach(findTags);
   634→    };
   635→    findTags(designTree);
   636→    const semanticCount = semanticTags.filter((tag) => usedTags.has(tag)).length;
   637→    report.semantics = Math.min(100, Math.round((semanticCount / 3) * 100)); // 3+ semantic tags = good
   638→
   639→    // Check responsiveness
   640→    const hasFlexOrGrid = /flex|grid/.test(styleStr);
   641→    const hasPercentOrRelative = /%|rem|vh|vw/.test(styleStr);
   642→    report.responsiveness = hasFlexOrGrid ? (hasPercentOrRelative ? 90 : 70) : 40;
   643→
   644→    // Check accessibility basics
   645→    const hasHeadings = /"heading"|"h1"|"h2"|"h3"/.test(styleStr);
   646→    const hasButtons = /"button"/.test(styleStr);
   647→    report.accessibility = hasHeadings ? (hasButtons ? 80 : 60) : 40;
   648→    if (!hasHeadings) {
   649→      report.issues.push({
   650→        severity: 'warning',
   651→        category: 'accessibility',
   652→        message: 'No heading elements found — important for screen readers',
   653→        autoFixable: false,
   654→      });
   655→    }
   656→
   657→    // Suggestions
   658→    if (report.responsiveness < 80) {
   659→      report.suggestions.push('Consider using flexbox/grid layouts with relative units for better responsiveness');
   660→    }
   661→    if (report.semantics < 80) {
   662→      report.suggestions.push('Use semantic HTML tags (header, nav, main, section, footer) for better structure');
   663→    }
   664→    if (report.accessibility < 80) {
   665→      report.suggestions.push('Add proper heading hierarchy and ARIA labels for accessibility');
   666→    }
   667→
   668→    // Overall score
   669→    report.overallScore = Math.round(
   670→      (report.completeness * 0.25 +
   671→        report.cssValid * 0.2 +
   672→        report.semantics * 0.2 +
   673→        report.responsiveness * 0.2 +
   674→        report.accessibility * 0.15)
   675→    );
   676→
   677→    return report;
   678→  }, []);
