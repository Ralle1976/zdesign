   375→// ============ Chat Response Parser ============
   376→
   377→/**
   378→ * Aggressively repair malformed JSON produced by LLMs.
   379→ * Handles: markdown wrappers, key="value" (equals), unquoted keys,
   380→ * trailing commas, single quotes, JS comments, extra text, unbalanced brackets.
   381→ * Returns the repaired string or null if unrepairable.
   382→ */
   383→export function repairLLMJson(text: string): string | null {
   384→  let s = text;
   385→
   386→  // 1. Remove markdown code block wrappers (```json ... ```)
   387→  const codeBlockMatch = s.match(/```(?:json)?\s*\n?([\s\S]*?)\n?\s*```/);
   388→  if (codeBlockMatch) {
   389→    s = codeBlockMatch[1];
   390→  }
   391→
   392→  // 2. Remove JavaScript-style comments
   393→  // Single-line comments (// ...)
   394→  s = s.replace(/\/\/.*$/gm, '');
   395→  // Multi-line comments (/* ... */)
   396→  s = s.replace(/\/\*[\s\S]*?\*\//g, '');
   397→
   398→  // 3. Remove any text before the first { and after the last }
   399→  const firstBrace = s.indexOf('{');
   400→  const lastBrace = s.lastIndexOf('}');
   401→  if (firstBrace === -1 || lastBrace === -1 || firstBrace > lastBrace) {
   402→    return null;
   403→  }
   404→  s = s.substring(firstBrace, lastBrace + 1);
   405→
   406→  // 4. Fix "key: "value"" pattern → "key": "value" (LLM often drops the closing quote before colon)
   407→  // This is the MOST COMMON LLM JSON error: the key starts with " but the closing " before : is missing
   408→  // Example: "maxWidth: "320px" → "maxWidth": "320px"
   409→  // We match this after { or , to avoid false positives inside value strings
   410→  s = s.replace(/([{,]\s*)"([a-zA-Z_]\w*)\s*:\s*"/g, '$1"$2": "');
   411→
   412→  // 4b. Fix content-after-object pattern (LLM places content string OUTSIDE the node object)
   413→  // This is the SECOND MOST COMMON LLM error: content values appear as separate array elements
   414→  // The LLM writes: {"id": "x", "style": {...}}, "Content Text"},
   415→  // Should be:       {"id": "x", "style": {...}, "content": "Content Text"},
   416→  //
   417→  // The pattern is: double-close-brace, comma, string, then comma/close-bracket/parent-close
   418→  // We need to: remove one close-brace, add "content" property, close the node, keep the separator
   419→  // Note: (?:[^"\\]|\\.)* handles escaped quotes inside strings (e.g., "He said \"hello\"")
   420→  
   421→  // Fix: }}, "STRING"}, → }, "content": "STRING"},
   422→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(,)/g, '}, "content": "$1"$2');
   423→  // Fix: }}, "STRING"}] → }, "content": "STRING"}]
   424→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\])/g, '}, "content": "$1"}$2');
   425→  // Fix: }}, "STRING"} followed by another { → }, "content": "STRING"}, {
   426→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\{)/g, '}, "content": "$1"}, $2');
   427→  // Fix: }}, "STRING"}} → }, "content": "STRING"}}  (parent close after)
   428→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\}\})/g, '}, "content": "$1"}$2');
   429→  // Fix: }}, "STRING"}  (end of input or other)
   430→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\})/g, '}, "content": "$1"$2');
   431→
   432→  // 5. Fix key="value" patterns → "key": "value"
   433→  // Handles: key="value", key = "value", key= "value", key ="value"
   434→  s = s.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*=\s*"/g, '$1"$2": "');
   435→
   436→  // 6. Fix key='value' patterns → "key": "value"
   437→  s = s.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*=\s*'/g, '$1"$2": "');
   438→
   439→  // 7. Fix unquoted property names (e.g., {display: "flex"} → {"display": "flex"})
   440→  // Match: after { or , followed by an identifier followed by :
   441→  // But be careful not to re-quote already-quoted keys
   442→  s = s.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*:/g, (match, prefix, key) => {
   443→    // Check if the key is already quoted (shouldn't match in that case, but just in case)
   444→    if (prefix.endsWith('"') || prefix.endsWith("'")) return match;
   445→    return `${prefix}"${key}":`;
   446→  });
   447→
   448→  // 7. Fix single quotes used as string delimiters (not apostrophes inside strings)
   449→  // Only replace single quotes that appear to be string delimiters:
   450→  // - After : or [ or , followed by ' (opening quote)
   451→  // - Before : or ] or , preceded by ' (closing quote)
   452→  // Do NOT blindly replace all ' with " as it breaks strings like "I'll"
   453→  // Since the LLM typically uses double quotes already, skip this step if no single-quote delimiters found
   454→  if (/[:\[,]\s*'[^']*'\s*[:,\]}]/.test(s)) {
   455→    // Replace opening single quotes after : [ ,
   456→    s = s.replace(/([:\[,]\s*)'/g, '$1"');
   457→    // Replace closing single quotes before : , ] }
   458→    s = s.replace(/'(\s*[:,\]}])/g, '"$1');
   459→  }
   460→
   461→  // 7c. Fix spaces in rgba/hsl function values (LLM often writes "rgba(0, 0, 0, 0.1)" instead of "rgba(0,0,0,0.1)")
   462→  // This matters because the LLM sometimes outputs unquoted values like: "boxShadow": 0 4px 6px rgba(0, 0, 0, 0.1)
   463→  // and the spaces in rgba() cause parse issues when we try to quote the value later
   464→  s = s.replace(/rgba\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)\s*,\s*([\d.]+)\s*\)/g, 'rgba($1,$2,$3,$4)');
   465→  s = s.replace(/hsl\(\s*(\d+)\s*,\s*(\d+)%\s*,\s*(\d+)%\s*\)/g, 'hsl($1,$2%,$3%)');
   466→  s = s.replace(/hsla\(\s*(\d+)\s*,\s*(\d+)%\s*,\s*(\d+)%\s*,\s*([\d.]+)\s*\)/g, 'hsla($1,$2%,$3%,$4)');
   467→
   468→  // 7d. Fix unquoted CSS-like values after known CSS property names
   469→  // The LLM sometimes writes: "boxShadow": 0 4px 6px rgba(0,0,0,0.1) (missing quotes around value)
   470→  // or: "transform": rotate(45deg) (missing quotes around value)
   471→  // Pattern: after a quoted CSS property name and colon, if the value starts with a digit, "rgba(", "hsl", "rotate", "translate", "scale", "linear-gradient", etc.
   472→  // and is NOT already quoted, wrap it in quotes.
   473→  // We look for: "cssProp": <unquoted-value> where unquoted-value continues until , } or ]
   474→  const cssPropsWithUnquotedValues = [
   475→    'boxShadow', 'textShadow', 'filter', 'backdropFilter', 'transform',
   476→    'transition', 'animation', 'backgroundImage', 'background',
   477→    'border', 'borderTop', 'borderRight', 'borderBottom', 'borderLeft',
   478→    'outline', 'margin', 'padding', 'borderRadius',
   479→  ];
   480→  for (const prop of cssPropsWithUnquotedValues) {
   481→    // Match: "prop": <value-not-starting-with-quote> up to comma, }, or ]
   482→    // The value must NOT start with " or { or [ or true/false/null/number
   483→    // We specifically target values that start with digits, letters, or rgba/hsl etc.
   484→    const propRegex = new RegExp(
   485→      `("${prop}")\\s*:\\s*(?!"|\\{|\\[|true|false|null)([a-zA-Z0-9().,\\s%#_-]+?)(\\s*[,}\\]])`,
   486→      'g'
   487→    );
   488→    s = s.replace(propRegex, '$1: "$2"$3');
   489→  }
   490→
   491→  // 8. Fix trailing commas before } or ]
   492→  s = s.replace(/,\s*([}\]])/g, '$1');
   493→
   494→  // 9. Try to balance brackets/braces
   495→  const openBraces = (s.match(/{/g) || []).length;
   496→  const closeBraces = (s.match(/}/g) || []).length;
   497→  const openBrackets = (s.match(/\[/g) || []).length;
   498→  const closeBrackets = (s.match(/]/g) || []).length;
   499→
   500→  // Add missing closing brackets/braces at the end
   501→  if (openBraces > closeBraces) {
   502→    s += '}'.repeat(openBraces - closeBraces);
   503→  }
   504→  if (openBrackets > closeBrackets) {
   505→    s += ']'.repeat(openBrackets - closeBrackets);
   506→  }
   507→
   508→  // 10. Fix missing commas between properties (heuristic)
   509→  // Pattern: "value" followed by whitespace then "key": — likely missing comma
   510→  s = s.replace(/("\s*)(\n\s*"[a-zA-Z_]\w*"\s*:)/g, '$1,$2');
   511→
   512→  // 11. Validate by attempting to parse
   513→  try {
   514→    JSON.parse(s);
   515→    return s;
   516→  } catch {
   517→    // If still invalid, try more aggressive fixes
   518→    try {
   519→      // Attempt progressive truncation from the end to find valid JSON
   520→      for (let len = s.length; len > 10; len -= 1) {
   521→        const truncated = s.substring(0, len);
   522→        // Count brackets to ensure balance
   523→        const ob = (truncated.match(/{/g) || []).length;
   524→        const cb = (truncated.match(/}/g) || []).length;
   525→        const obr = (truncated.match(/\[/g) || []).length;
   526→        const cbr = (truncated.match(/]/g) || []).length;
   527→        if (ob === cb && obr === cbr) {
   528→          try {
   529→            JSON.parse(truncated);
   530→            return truncated;
   531→          } catch {
   532→            // continue
   533→          }
   534→        }
   535→      }
   536→    } catch {
   537→      // Give up
   538→    }
   539→
   540→    // 12. Last resort: try to extract just the design object by finding the deepest valid JSON prefix
   541→    // This handles cases where the LLM generates a mostly-valid structure but has trailing garbage
   542→    try {
   543→      // Find the "design" key and try to extract its value
   544→      const designKeyIdx = s.indexOf('"design"');
   545→      if (designKeyIdx !== -1) {
   546→        // Find the opening brace of the design value
   547→        const designBraceIdx = s.indexOf('{', designKeyIdx + 8);
   548→        if (designBraceIdx !== -1) {
   549→          // Try to find a valid design object by scanning forward
   550→          let depth = 0;
   551→          let bestEnd = -1;
   552→          for (let i = designBraceIdx; i < s.length; i++) {
   553→            if (s[i] === '{') depth++;
   554→            else if (s[i] === '}') depth--;
   555→            if (depth === 0) {
   556→              // We found a balanced design object, try to parse it
   557→              const candidate = s.substring(designBraceIdx, i + 1);
   558→              try {
   559→                JSON.parse(candidate);
   560→                bestEnd = i;
   561→                // Don't break — keep looking for a larger valid object
   562→              } catch {
   563→                // This balanced section isn't valid JSON, continue
   564→              }
   565→            }
   566→          }
   567→          if (bestEnd !== -1) {
   568→            const designObj = s.substring(designBraceIdx, bestEnd + 1);
   569→            // Wrap in a response object
   570→            const wrapped = `{"message": "Design recovered from partial response", "design": ${designObj}}`;
   571→            try {
   572→              JSON.parse(wrapped);
   573→              return wrapped;
   574→            } catch {
   575→              // The design object alone might be valid
   576→              try {
   577→                JSON.parse(designObj);
   578→                return designObj;
   579→              } catch {
   580→                // continue
   581→              }
   582→            }
   583→          }
   584→        }
   585→      }
   586→    } catch {
   587→      // Give up
   588→    }
   589→
   590→    return null;
   591→  }
   592→}

   594→/**
   595→ * Try to parse JSON with aggressive error repair.
   596→ * Handles common LLM mistakes: unquoted keys, key="value", trailing commas,
   597→ * single quotes, JS comments, extra text around JSON.
   598→ */
   599→function tryParseJSON(text: string): unknown | null {
   600→  // First, try direct parse
   601→  try {
   602→    return JSON.parse(text);
   603→  } catch {
   604→    // continue to repair attempts
   605→  }
   606→
   607→  // Use the aggressive repair function
   608→  const repaired = repairLLMJson(text);
   609→  if (repaired !== null) {
   610→    try {
   611→      return JSON.parse(repaired);
   612→    } catch {
   613→      // continue
   614→    }
   615→  }
   616→
   617→  // Additional targeted fixes if repairLLMJson failed
   618→  let fixed = text;
   619→
   620→  // Fix 1: key="value" patterns (equals instead of colon)
   621→  fixed = fixed.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*=\s*"/g, '$1"$2": "');
   622→  try { return JSON.parse(fixed); } catch { /* continue */ }
   623→
   624→  // Fix 2: Unquoted property names
   625→  fixed = fixed.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*:/g, '$1"$2":');
   626→  try { return JSON.parse(fixed); } catch { /* continue */ }
   627→
   628→  // Fix 3: Trailing commas
   629→  fixed = fixed.replace(/,\s*([}\]])/g, '$1');
   630→  try { return JSON.parse(fixed); } catch { /* continue */ }
   631→
   632→  // Fix 4: Single quotes to double quotes
   633→  fixed = fixed.replace(/'/g, '"');
   634→  try { return JSON.parse(fixed); } catch { /* continue */ }
   635→
   636→  // Fix 5: Remove JS comments
   637→  fixed = fixed.replace(/\/\/.*$/gm, '');
   638→  fixed = fixed.replace(/\/\*[\s\S]*?\*\//g, '');
   639→  try { return JSON.parse(fixed); } catch { /* continue */ }
   640→
   641→  // Fix 6: Missing commas between properties
   642→  fixed = fixed.replace(/("\s*)(\n\s*"[a-zA-Z_]\w*"\s*:)/g, '$1,$2');
   643→  try { return JSON.parse(fixed); } catch { /* continue */ }
   644→
   645→  // Fix 7: Try balancing brackets
   646→  const ob = (fixed.match(/{/g) || []).length;
   647→  const cb = (fixed.match(/}/g) || []).length;
   648→  const obr = (fixed.match(/\[/g) || []).length;
   649→  const cbr = (fixed.match(/]/g) || []).length;
   650→  if (ob > cb) fixed += '}'.repeat(ob - cb);
   651→  if (obr > cbr) fixed += ']'.repeat(obr - cbr);
   652→  try { return JSON.parse(fixed); } catch { /* continue */ }
   653→
   654→  return null;
   655→}

   657→export function parseAIResponse(response: string): {
   658→  message: string;
   659→  design: unknown | null;
   660→  parseFailed?: boolean;
   661→} {
   662→  // Helper: extract design from parsed object
   663→  function extractDesign(parsed: Record<string, unknown>): { message: string; design: unknown | null } | null {
   664→    if (parsed.design || parsed.id || parsed.type || parsed.children) {
   665→      return {
   666→        message: (parsed.message as string) || 'Design generated successfully.',
   667→        design: parsed.design || parsed,
   668→      };
   669→    }
   670→    return null;
   671→  }
   672→
   673→  // 1. Try to parse the full response as JSON
   674→  const directParsed = tryParseJSON(response);
   675→  if (directParsed && typeof directParsed === 'object') {
   676→    const extracted = extractDesign(directParsed as Record<string, unknown>);
   677→    if (extracted) return extracted;
   678→  }
   679→
   680→  // 2. Try aggressive repair on the full response
   681→  const repaired = repairLLMJson(response);
   682→  if (repaired) {
   683→    try {
   684→      const repairedParsed = JSON.parse(repaired);
   685→      if (repairedParsed && typeof repairedParsed === 'object') {
   686→        const extracted = extractDesign(repairedParsed as Record<string, unknown>);
   687→        if (extracted) return extracted;
   688→      }
   689→    } catch {
   690→      // continue
   691→    }
   692→  }
   693→
   694→  // 3. Try to extract JSON from markdown code blocks (possibly multiple)
   695→  const codeBlockRegex = /```(?:json)?\s*\n?([\s\S]*?)\n?```/g;
   696→  let match;
   697→  while ((match = codeBlockRegex.exec(response)) !== null) {
   698→    const parsed = tryParseJSON(match[1]);
   699→    if (parsed && typeof parsed === 'object') {
   700→      const extracted = extractDesign(parsed as Record<string, unknown>);
   701→      if (extracted) {
   702→        // If there's text before the code block, use it as the message
   703→        const beforeCode = response.substring(0, match.index).trim();
   704→        if (beforeCode) {
   705→          extracted.message = beforeCode.replace(/^[^\w]*/, '').trim() || extracted.message;
   706→        }
   707→        return extracted;
   708→      }
   709→    }
   710→  }
   711→
   712→  // 4. Try to find the largest JSON object in the response
   713→  const objectMatches = response.match(/\{[\s\S]*"id"[\s\S]*\}/g);
   714→  if (objectMatches) {
   715→    // Try from largest to smallest
   716→    const sorted = objectMatches.sort((a, b) => b.length - a.length);
   717→    for (const objMatch of sorted) {
   718→      const parsed = tryParseJSON(objMatch);
   719→      if (parsed && typeof parsed === 'object') {
   720→        const extracted = extractDesign(parsed as Record<string, unknown>);
   721→        if (extracted) return extracted;
   722→      }
   723→    }
   724→  }
   725→
   726→  // 5. Try to find any JSON-like object with "id" (using unquoted key search too)
   727→  const looseObjectMatches = response.match(/\{[\s\S]*(?:["']?id["']?\s*[:=])[\s\S]*\}/g);
   728→  if (looseObjectMatches) {
   729→    const sorted = looseObjectMatches.sort((a, b) => b.length - a.length);
   730→    for (const objMatch of sorted) {
   731→      const parsed = tryParseJSON(objMatch);
   732→      if (parsed && typeof parsed === 'object') {
   733→        const extracted = extractDesign(parsed as Record<string, unknown>);
   734→        if (extracted) return extracted;
   735→      }
   736→    }
   737→  }
   738→
   739→  // 6. Return as plain text message (strip markdown code blocks for cleaner output)
   740→  const cleanedMessage = response
   741→    .replace(/```(?:json)?\s*\n?/g, '')
   742→    .replace(/\n?```/g, '')
   743→    .trim();
   744→  return { message: cleanedMessage || response, design: null, parseFailed: true };
   745→}
