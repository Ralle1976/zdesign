    70→function GenerationStepProgress() {

    72→  const generationProgress = useZDesignStore((s) => s.generationProgress);
    73→  const creativeMode = useZDesignStore((s) => s.creativeMode);
    74→  const isDe = locale === 'de';
    75→

    78→    { id: 'generating', label: isDe ? 'Generiere Layout...' : 'Generating layout...', icon: Sparkles },
    79→    { id: 'rendering', label: isDe ? 'Füge Stile hinzu...' : 'Adding styles...', icon: Palette },
    80→    { id: 'evaluating', label: isDe ? 'Finalisiere...' : 'Finalizing design...', icon: CheckCircle2 },
    81→  ];
    82→
    83→  const currentIdx = steps.findIndex((s) => s.id === generationProgress.stage);
    84→
    85→  return (
    86→    <div className="px-4 py-2 space-y-2">
    87→      {/* Main status line */}
    88→      <div className="flex items-center gap-2">
    89→        <Loader2 className="size-3.5 animate-spin text-emerald-500 shrink-0" />
    90→        <span className="text-xs font-medium text-foreground">
    91→          {isDe ? 'KI generiert Design...' : 'Generating design with AI...'}
    92→        </span>
    93→        {generationProgress.estimatedTimeLeft && generationProgress.estimatedTimeLeft > 0 && (
    94→          <span className="text-[10px] text-muted-foreground ml-auto">
    95→            ~{generationProgress.estimatedTimeLeft}s {isDe ? 'übrig' : 'remaining'}
    96→          </span>
    97→        )}
    98→      </div>
    99→
   100→      {/* Progress bar */}
   101→      <Progress value={generationProgress.percentage} className="h-1.5 [&>div]:bg-emerald-500" />
   102→
   103→      {/* Steps */}
   104→      <div className="flex items-center gap-1">
   105→        {steps.map((step, idx) => {

   383→  const clearChat = useZDesignStore((s) => s.clearChat);

   385→  const setIsGenerating = useZDesignStore((s) => s.setIsGenerating);
   386→  const projectId = useZDesignStore((s) => s.projectId);
   387→  const designTree = useZDesignStore((s) => s.designTree);
   388→  const designSystem = useZDesignStore((s) => s.designSystem);
   389→  const setDesignTree = useZDesignStore((s) => s.setDesignTree);
   390→  const setGenerationProgress = useZDesignStore((s) => s.setGenerationProgress);
   391→  const resetGenerationProgress = useZDesignStore((s) => s.resetGenerationProgress);
   392→  const setQualityReport = useZDesignStore((s) => s.setQualityReport);
   393→  const qualityReport = useZDesignStore((s) => s.qualityReport);
   394→  const generationProgress = useZDesignStore((s) => s.generationProgress);
   395→
   396→  const [input, setInput] = useState('');
   397→  const [isEnhancing, setIsEnhancing] = useState(false);
   398→  const scrollRef = useRef<HTMLDivElement>(null);
   399→  const textareaRef = useRef<HTMLTextAreaElement>(null);

   401→  // AI Image dialog
   402→  const [aiImageOpen, setAiImageOpen] = useState(false);
   403→
   404→  // Research dialog
   405→  const [researchOpen, setResearchOpen] = useState(false);
   406→
   407→  // === Refs for mutable state to avoid stale closures ===
   408→  const projectIdRef = useRef(projectId);
   409→  const isGeneratingRef = useRef(isGenerating);
   410→  const chatMessagesRef = useRef(chatMessages);
   411→  const designTreeRef = useRef(designTree);
   412→  const designSystemRef = useRef(designSystem);
   413→
   414→  useEffect(() => { projectIdRef.current = projectId; }, [projectId]);
   415→  useEffect(() => { isGeneratingRef.current = isGenerating; }, [isGenerating]);
   416→  useEffect(() => { chatMessagesRef.current = chatMessages; }, [chatMessages]);
   417→  useEffect(() => { designTreeRef.current = designTree; }, [designTree]);
   418→  useEffect(() => { designSystemRef.current = designSystem; }, [designSystem]);
   419→
   420→  // Stable store action refs
   421→  const addChatMessageRef = useRef(addChatMessage);
   422→  const setIsGeneratingRef = useRef(setIsGenerating);
   423→  const setDesignTreeRef = useRef(setDesignTree);
   424→  const setGenerationProgressRef = useRef(setGenerationProgress);
   425→  const resetGenerationProgressRef = useRef(resetGenerationProgress);
   426→  const setQualityReportRef = useRef(setQualityReport);
   427→  const tRef = useRef(t);
   428→
   429→  useEffect(() => { addChatMessageRef.current = addChatMessage; }, [addChatMessage]);
   430→  useEffect(() => { setIsGeneratingRef.current = setIsGenerating; }, [setIsGenerating]);
   431→  useEffect(() => { setDesignTreeRef.current = setDesignTree; }, [setDesignTree]);
   432→  useEffect(() => { setGenerationProgressRef.current = setGenerationProgress; }, [setGenerationProgress]);
   433→  useEffect(() => { resetGenerationProgressRef.current = resetGenerationProgress; }, [resetGenerationProgress]);
   434→  useEffect(() => { setQualityReportRef.current = setQualityReport; }, [setQualityReport]);
   435→  useEffect(() => { tRef.current = t; }, [t]);
   436→
   437→  // ============ Progress simulation ============
   438→  // Simulates step-by-step progress while the API call is in flight
   439→  // LLM takes ~60-90s, so we simulate progress over that duration
   440→  const progressTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);
   441→
   442→  const startProgressSimulation = useCallback(() => {
   443→    // Only set up incremental percentage updates during the API call
   444→    // Initial stage is already set by sendMessage
   445→    const startTime = Date.now();
   446→    const totalDuration = 80000; // 80 seconds estimated total for LLM generation
   447→
   448→    // Auto-advance stages on a timer
   449→    // analyzing: 0-5s, generating: 5-55s, rendering: handled after API returns
   450→    const stageTimers: Array<{ delay: number; stage: string; pct: number; msg: string }> = [
   451→      { delay: 5000, stage: 'generating', pct: 25, msg: 'AI is creating your design...' },
   452→      { delay: 20000, stage: 'generating', pct: 35, msg: 'Building layout structure...' },
   453→      { delay: 40000, stage: 'generating', pct: 50, msg: 'Adding components and styles...' },
   454→      { delay: 60000, stage: 'generating', pct: 60, msg: 'Polishing design details...' },
   455→    ];
   456→
   457→    const stageTimeouts: Array<ReturnType<typeof setTimeout>> = [];
   458→    for (const st of stageTimers) {
   459→      stageTimeouts.push(setTimeout(() => {
   460→        setGenerationProgressRef.current({
   461→          stage: st.stage as 'analyzing' | 'generating' | 'rendering' | 'evaluating' | 'complete' | 'error' | 'idle',
   462→          stageLabel: st.msg,
   463→          percentage: st.pct,
   464→          message: st.msg,
   465→          estimatedTimeLeft: Math.max(0, Math.round((totalDuration - st.delay) / 1000)),
   466→        });
   467→      }, st.delay));
   468→    }
   469→
   470→    progressTimerRef.current = setInterval(() => {
   471→      const elapsed = Date.now() - startTime;
   472→      const pct = Math.min(65, Math.round(20 + (elapsed / totalDuration) * 45)); // 20-65% range
   473→      const timeLeft = Math.max(0, Math.round((totalDuration - elapsed) / 1000));
   474→
   475→      // Update percentage and time remaining
   476→      setGenerationProgressRef.current({
   477→        percentage: pct,
   478→        estimatedTimeLeft: timeLeft,
   479→      });
   480→    }, 1000);
   481→
   482→    // Store stage timeouts for cleanup
   483→    (progressTimerRef as React.MutableRefObject<ReturnType<typeof setInterval> | null & { stageTimeouts?: Array<ReturnType<typeof setTimeout>> }>).current = {
   484→      ...progressTimerRef.current,
   485→      stageTimeouts,
   486→    } as unknown as ReturnType<typeof setInterval>;
   487→  }, []);
   488→
   489→  const stopProgressSimulation = useCallback((success: boolean) => {
   490→    if (progressTimerRef.current) {
   491→      clearInterval(progressTimerRef.current);
   492→      // Clean up stage timeouts
   493→      const ref = progressTimerRef.current as unknown as { stageTimeouts?: Array<ReturnType<typeof setTimeout>> };
   494→      if (ref.stageTimeouts) {
   495→        ref.stageTimeouts.forEach(clearTimeout);
   496→      }
   497→      progressTimerRef.current = null;
   498→    }
   499→
   500→    if (success) {
   501→      // If we're still in evaluating stage from sendMessage, set complete
   502→      // If we haven't reached evaluating yet, skip to complete
   503→      setGenerationProgressRef.current({
   504→        stage: 'complete',
   505→        stageLabel: 'Complete!',

   528→  // ============ Evaluate Design Quality ============
   529→  const evaluateDesignQuality = useCallback((designTree: Record<string, unknown>) => {
   530→    const report = {
   531→      overallScore: 0,
   532→      completeness: 0,
   533→      cssValid: 0,
   534→      semantics: 0,
   535→      responsiveness: 0,
   536→      accessibility: 0,
   537→      issues: [] as Array<{
   538→        severity: 'critical' | 'warning' | 'info';
   539→        category: 'css' | 'semantics' | 'responsiveness' | 'accessibility' | 'completeness';
   540→        message: string;
   541→        autoFixable: boolean;
   542→      }>,
   543→      suggestions: [] as string[],
   544→    };
   545→
   546→    // Check completeness
   547→    const children = (designTree as { children?: unknown[] }).children;
   548→    if (children && children.length > 0) {
   549→      report.completeness = Math.min(100, children.length * 20);
   550→      // Check for essential sections
   551→      const hasNav = children.some((c: unknown) => {
   552→        const node = c as { type?: string; tag?: string };
   553→        return node.type === 'nav' || node.tag === 'nav' || node.type === 'header';
   554→      });
   555→      const hasFooter = children.some((c: unknown) => {
   556→        const node = c as { type?: string; tag?: string };
   557→        return node.type === 'footer' || node.tag === 'footer';
   558→      });
   559→      const hasContent = children.some((c: unknown) => {
   560→        const node = c as { type?: string; tag?: string };
   561→        return node.type === 'section' || node.tag === 'section' || node.type === 'main';
   562→      });
   563→
   564→      if (!hasNav) {
   565→        report.issues.push({
   566→          severity: 'warning',
   567→          category: 'completeness',
   568→          message: 'No navigation section found',
   569→          autoFixable: false,
   570→        });
   571→      }
   572→      if (!hasFooter) {
   573→        report.issues.push({
   574→          severity: 'info',
   575→          category: 'completeness',
   576→          message: 'No footer section found',
   577→          autoFixable: false,
   578→        });
   579→      }
   580→      if (!hasContent) {
   581→        report.issues.push({
   582→          severity: 'warning',
   583→          category: 'completeness',
   584→          message: 'No main content section found',
   585→          autoFixable: false,
   586→        });
   587→      }
   588→
   589→      if (hasNav && hasFooter && hasContent) {
   590→        report.completeness = 100;
   591→      }
   592→    } else {
   593→      report.issues.push({
   594→        severity: 'critical',
   595→        category: 'completeness',
   596→        message: 'Design tree has no children',
   597→        autoFixable: false,
   598→      });
   599→    }
   600→
   601→    // Check CSS validity (basic check)
   602→    const styleStr = JSON.stringify(designTree);
   603→    const hasTailwindShorthand = /\d+xl|text-md|p-lg|gap-md/.test(styleStr);
   604→    if (hasTailwindShorthand) {
   605→      report.cssValid = 50;
   606→      report.issues.push({
   607→        severity: 'critical',
   608→        category: 'css',
   609→        message: 'Found Tailwind shorthand in CSS values — should use real CSS (px, rem, %)',
   610→        autoFixable: true,
   611→      });
   612→    } else {
   613→      report.cssValid = 95;
   614→    }
   615→
   616→    // Check semantics
   617→    const semanticTags = ['header', 'nav', 'main', 'section', 'footer', 'article', 'aside'];
   618→    const usedTags = new Set<string>();
   619→    const findTags = (node: unknown) => {
   620→      const n = node as { tag?: string; children?: unknown[] };
   621→      if (n.tag) usedTags.add(n.tag);
   622→      if (n.children) n.children.forEach(findTags);
   623→    };
   624→    findTags(designTree);
   625→    const semanticCount = semanticTags.filter((tag) => usedTags.has(tag)).length;
   626→    report.semantics = Math.min(100, Math.round((semanticCount / 3) * 100)); // 3+ semantic tags = good
   627→
   628→    // Check responsiveness
   629→    const hasFlexOrGrid = /flex|grid/.test(styleStr);
   630→    const hasPercentOrRelative = /%|rem|vh|vw/.test(styleStr);
   631→    report.responsiveness = hasFlexOrGrid ? (hasPercentOrRelative ? 90 : 70) : 40;
   632→
   633→    // Check accessibility basics
   634→    const hasHeadings = /"heading"|"h1"|"h2"|"h3"/.test(styleStr);
   635→    const hasButtons = /"button"/.test(styleStr);
   636→    report.accessibility = hasHeadings ? (hasButtons ? 80 : 60) : 40;
   637→    if (!hasHeadings) {
   638→      report.issues.push({
   639→        severity: 'warning',
   640→        category: 'accessibility',
   641→        message: 'No heading elements found — important for screen readers',
   642→        autoFixable: false,
   643→      });
   644→    }
   645→
   646→    // Suggestions
   647→    if (report.responsiveness < 80) {
   648→      report.suggestions.push('Consider using flexbox/grid layouts with relative units for better responsiveness');
   649→    }
   650→    if (report.semantics < 80) {
   651→      report.suggestions.push('Use semantic HTML tags (header, nav, main, section, footer) for better structure');
   652→    }
   653→    if (report.accessibility < 80) {
   654→      report.suggestions.push('Add proper heading hierarchy and ARIA labels for accessibility');
   655→    }
   656→
   657→    // Overall score
   658→    report.overallScore = Math.round(
   659→      (report.completeness * 0.25 +
   660→        report.cssValid * 0.2 +
   661→        report.semantics * 0.2 +
   662→        report.responsiveness * 0.2 +
   663→        report.accessibility * 0.15)
   664→    );
   665→
   666→    return report;
   667→  }, []);
