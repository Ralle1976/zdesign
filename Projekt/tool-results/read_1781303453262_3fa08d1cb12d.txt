   234→// ============ Helper Functions ============
   235→
   236→function getRateLimitColor(used: number, limit: number): string {
   237→  if (limit <= 0) return 'bg-muted-foreground/30';
   238→  const pct = used / limit;
   239→  if (pct > 0.75) return 'bg-red-500';
   240→  if (pct > 0.5) return 'bg-amber-500';
   241→  return 'bg-emerald-500';
   242→}
   243→
   244→function getRateLimitTextColor(used: number, limit: number): string {
   245→  if (limit <= 0) return 'text-muted-foreground';
   246→  const pct = used / limit;
   247→  if (pct > 0.75) return 'text-red-600 dark:text-red-400';
   248→  if (pct > 0.5) return 'text-amber-600 dark:text-amber-400';
   249→  return 'text-emerald-600 dark:text-emerald-400';
   250→}
   251→
   252→function formatNumber(n: number): string {
   253→  if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(1)}M`;
   254→  if (n >= 1_000) return `${(n / 1_000).toFixed(1)}K`;
   255→  return n.toString();
   256→}
   257→
   258→function formatResetTime(dateStr: string | null | undefined): string {
   259→  if (!dateStr) return '';
   260→  const date = new Date(dateStr);
   261→  const now = new Date();
   262→  const diffMs = date.getTime() - now.getTime();
   263→  if (diffMs <= 0) return 'Reset available';
   264→  const diffMin = Math.floor(diffMs / 60000);
   265→  if (diffMin < 60) return `Resets in ${diffMin}m`;
   266→  const diffHr = Math.floor(diffMin / 60);
   267→  return `Resets in ${diffHr}h ${diffMin % 60}m`;
   268→}
   269→
   270→function formatContextLength(ctx?: number): string {
   271→  if (!ctx) return '-';
   272→  if (ctx >= 1_000_000) return `${(ctx / 1_000_000).toFixed(0)}M`;
   273→  if (ctx >= 1_000) return `${(ctx / 1_000).toFixed(0)}K`;
   274→  return ctx.toString();
   275→}
   276→
   277→function formatPricing(pricing?: DynamicModelInfo['pricing']): string {
   278→  if (!pricing) return '-';
   279→  const parts: string[] = [];
   280→  if (pricing.prompt && pricing.prompt > 0) {
   281→    parts.push(`In: $${pricing.prompt.toFixed(2)}/M`);
   282→  }
   283→  if (pricing.completion && pricing.completion > 0) {
   284→    parts.push(`Out: $${pricing.completion.toFixed(2)}/M`);
   285→  }
   286→  if (parts.length === 0) return '-';
   287→  return parts.join(' · ');
   288→}

   389→      setTestingId(null);

  1699→      {providersWithLimits.length === 0 && (
