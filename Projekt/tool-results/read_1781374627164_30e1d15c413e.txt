   975→    let userContent = message;
   976→    if (designTree && (designTree as { children?: unknown[] }).children?.length as number > 0) {
   977→      const childrenCount = (designTree as { children: unknown[] }).children.length;
   978→      userContent = `[Current design exists with ${childrenCount} top-level sections. The user wants to modify it.]\n\nUser request: ${message}\n\nReturn the COMPLETE updated design tree (not just the changes).`;
   979→    }
   980→
   981→    // Build conversation messages from history (up to 4 most recent to save tokens)
   982→    const messages: Array<{ role: 'system' | 'user' | 'assistant'; content: string }> = [
   983→      { role: 'system', content: systemPrompt },
   984→    ];
   985→
   986→    if (history && history.length > 0) {
   987→      const recentHistory = history.slice(-4);
   988→      for (const msg of recentHistory) {
   989→        messages.push({ role: msg.role, content: msg.content });
   990→      }
   991→    }
   992→
   993→    messages.push({ role: 'user', content: userContent });
   994→
   995→    // Call Z.ai LLM with timeout and retry logic
   996→    let rawResponse: string | null = null;
   997→    let tokensUsed = 0;
   998→    let usedFallback = false;
   999→
  1000→    // Timeout strategy: LLM takes ~60-90s to generate a full design, so we need generous timeouts
  1001→    // First attempt gets 120s, subsequent attempts get 90s — the LLM is slow but reliable
  1002→    const LLM_TIMEOUT_FIRST = 120000; // 120s for first attempt (LLM takes ~74s on average)
  1003→    const LLM_TIMEOUT_RETRY = 90000; // 90s for retries
  1004→    const maxRetries = 2; // 2 retries (3 total attempts) — the LLM is slow but reliable
  1005→
  1006→    // ALWAYS try the LLM first — it produces much better designs than fallbacks
  1007→    for (let attempt = 0; attempt <= maxRetries; attempt++) {
  1008→      try {
  1009→        const zai = await getZAI();
  1010→        const timeoutMs = attempt === 0 ? LLM_TIMEOUT_FIRST : LLM_TIMEOUT_RETRY;
  1011→
  1012→        // Race the LLM call against a timeout
  1013→        const completionPromise = zai.chat.completions.create({
  1014→          messages,
  1015→          thinking: { type: 'disabled' },
  1016→        });
  1017→
  1018→        const timeoutPromise = new Promise<never>((_, reject) =>
  1019→          setTimeout(() => reject(new Error('LLM timeout')), timeoutMs)
  1020→        );
  1021→
  1022→        const completion = await Promise.race([completionPromise, timeoutPromise]);
  1023→        rawResponse = completion.choices[0]?.message?.content || null;
  1024→        tokensUsed = completion.usage?.total_tokens || 0;
  1025→        break; // Success, exit retry loop
  1026→      } catch (llmError) {
  1027→        console.warn(`[Chat API] Z.ai LLM attempt ${attempt + 1} failed:`, llmError instanceof Error ? llmError.message : llmError);
  1028→        if (attempt < maxRetries) {
  1029→          await new Promise((resolve) => setTimeout(resolve, 500));
  1030→        } else {
  1031→          console.error('[Chat API] Z.ai LLM attempts failed, using fallback');
  1032→          usedFallback = true;
  1033→        }
  1034→      }
  1035→    }
  1036→
  1037→    // Parse response: use LLM output or fallback
  1038→    let parsed: { message: string; design: unknown | null; parseFailed?: boolean };
  1039→    let templateUsed = false;
  1040→
  1041→    if (rawResponse) {
  1042→      parsed = parseAIResponse(rawResponse);
  1043→      if (!parsed.design) {
  1044→        console.warn('[Chat API] LLM response parsed but no design extracted. Raw length:', rawResponse.length, 'First 200:', rawResponse.substring(0, 200));
  1045→        // Try repairLLMJson directly
  1046→        const repaired = repairLLMJson(rawResponse);
  1047→        if (repaired) {
  1048→          try {
  1049→            const repairedParsed = JSON.parse(repaired);
  1050→            if (repairedParsed && typeof repairedParsed === 'object') {
  1051→              const rp = repairedParsed as Record<string, unknown>;
  1052→              if (rp.design || rp.id || rp.type || rp.children) {
  1053→                parsed = {
  1054→                  message: (rp.message as string) || parsed.message,
  1055→                  design: rp.design || rp,
  1056→                };
  1057→                console.log('[Chat API] Design recovered via repairLLMJson!');
  1058→              }
  1059→            }
  1060→          } catch (repairError) {
  1061→            const errMsg = repairError instanceof Error ? repairError.message : String(repairError);
  1062→            const posMatch = errMsg.match(/position (\d+)/);
  1063→            const pos = posMatch ? parseInt(posMatch[1]) : 0;
  1064→            console.warn('[Chat API] repairLLMJson also failed:', errMsg.substring(0, 100));
  1065→            if (pos > 0) {
  1066→              console.warn('[Chat API] Error context (pos', pos, '):', repaired.substring(Math.max(0, pos - 80), pos + 80));
  1067→            }
  1068→          }
  1069→        } else {
  1070→          console.warn('[Chat API] repairLLMJson returned null - unable to repair');
  1071→        }
  1072→
  1073→        // After repairLLMJson also fails, use smart contextual fallback
  1074→        if (!parsed.design && rawResponse) {
  1075→          // Extract the message from the LLM response to understand intent
  1076→          const messageMatch = rawResponse.match(/"message"\s*:\s*"([^"]+)"/);
  1077→          const llmMessage = messageMatch ? messageMatch[1] : '';
  1078→
  1079→          console.log('[Chat API] Using smart contextual fallback. User message:', message.substring(0, 60), 'LLM message:', llmMessage.substring(0, 60));
  1080→
  1081→          // Generate a contextual fallback based on the user's message AND the LLM's description
  1082→          const contextualFallback = generateContextualFallback(message, llmMessage, creativeMode);
  1083→          parsed = parseAIResponse(contextualFallback);
  1084→
  1085→          // Mark this as a template-based design
  1086→          templateUsed = true;
  1087→        }
  1088→      } else {
  1089→        console.log('[Chat API] Design parsed successfully!');
  1090→      }
  1091→    } else {
  1092→      // Fallback to hardcoded design generation (LLM was completely unavailable)
  1093→      const fallbackRaw = generateFallbackDesign(message, creativeMode);
  1094→      parsed = parseAIResponse(fallbackRaw);
  1095→      usedFallback = true;
  1096→    }
  1097→
  1098→    // Store assistant message
  1099→    const assistantMessage = await db.chatMessage.create({
  1100→      data: {
  1101→        projectId,
  1102→        role: 'assistant',
  1103→        content: parsed.message || 'I generated a design for you!',
  1104→        metadata: JSON.stringify({
  1105→          designUpdate: parsed.design || null,
  1106→          tokensUsed,
  1107→          usedFallback,
  1108→          creativeMode: creativeMode || false,
  1109→        }),
  1110→      },
  1111→    });
  1112→
  1113→    // Update project if design was generated
  1114→    if (parsed.design) {
  1115→      await db.project.update({
  1116→        where: { id: projectId },
  1117→        data: { designJSON: JSON.stringify(parsed.design), status: 'IN_PROGRESS' },
  1118→      });
  1119→    }
  1120→
  1121→    return NextResponse.json({
  1122→      id: assistantMessage.id,
  1123→      message: parsed.message || 'I generated a design for you!',
  1124→      design: parsed.design,
  1125→      projectId,
  1126→      createdAt: assistantMessage.createdAt,
  1127→      usedFallback,
  1128→      templateUsed,
  1129→      parseFailed: !parsed.design,
  1130→    });
  1131→  } catch (error) {
  1132→    console.error('[Chat API] Error:', error);
  1133→    return NextResponse.json(
  1134→      { error: 'Failed to process chat message', details: error instanceof Error ? error.message : 'Unknown error' },
  1135→      { status: 500 }
  1136→    );
  1137→  }
  1138→}

  1146→    if (!projectId) {
  1147→      return NextResponse.json({ error: 'projectId is required' }, { status: 400 });
  1148→    }
  1149→    const messages = await db.chatMessage.findMany({ where: { projectId }, orderBy: { createdAt: 'asc' } });
  1150→    return NextResponse.json({ messages });
  1151→  } catch (error) {
  1152→    console.error('[Chat API] GET Error:', error);
  1153→    return NextResponse.json({ error: 'Failed to retrieve chat history' }, { status: 500 });
  1154→  }
