   702→   702→            }

   704→   704→          }

   721→   721→        role: 'assistant',

   732→   732→    // Update project if design was generated

   752→   752→      { error: 'Failed to process chat message', details: error instanceof Error ? error.message : 'Unknown error' },
   753→   753→      { status: 500 }
   754→   754→    );
   755→   755→  }
