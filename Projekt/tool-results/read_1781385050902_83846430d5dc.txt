   232→   232→// ============ Chat Response Parser ============
   233→   233→
   234→   234→/**
   235→   235→ * Aggressively repair malformed JSON produced by LLMs.
   236→   236→ * Handles: markdown wrappers, key="value" (equals), unquoted keys,
   237→   237→ * trailing commas, single quotes, JS comments, extra text, unbalanced brackets.
   238→   238→ * Returns the repaired string or null if unrepairable.
   239→   239→ */
   240→   240→export function repairLLMJson(text: string): string | null {
   241→   241→  let s = text;
   242→   242→
   243→   243→  // 1. Remove markdown code block wrappers (```json ... ```)
   244→   244→  const codeBlockMatch = s.match(/```(?:json)?\s*\n?([\s\S]*?)\n?\s*```/);
   245→   245→  if (codeBlockMatch) {
   246→   246→    s = codeBlockMatch[1];
   247→   247→  }
   248→   248→
   249→   249→  // 2. Remove JavaScript-style comments
   250→   250→  // Single-line comments (// ...)
   251→   251→  s = s.replace(/\/\/.*$/gm, '');
   252→   252→  // Multi-line comments (/* ... */)
   253→   253→  s = s.replace(/\/\*[\s\S]*?\*\//g, '');
   254→   254→
   255→   255→  // 3. Remove any text before the first { and after the last }
   256→   256→  const firstBrace = s.indexOf('{');
   257→   257→  const lastBrace = s.lastIndexOf('}');
   258→   258→  if (firstBrace === -1 || lastBrace === -1 || firstBrace > lastBrace) {
   259→   259→    return null;
   260→   260→  }
   261→   261→  s = s.substring(firstBrace, lastBrace + 1);
   262→   262→
   263→   263→  // 4. Fix "key: "value"" pattern → "key": "value" (LLM often drops the closing quote before colon)
   264→   264→  // This is the MOST COMMON LLM JSON error: the key starts with " but the closing " before : is missing
   265→   265→  // Example: "maxWidth: "320px" → "maxWidth": "320px"
   266→   266→  // We match this after { or , to avoid false positives inside value strings
   267→   267→  s = s.replace(/([{,]\s*)"([a-zA-Z_]\w*)\s*:\s*"/g, '$1"$2": "');
   268→   268→
   269→   269→  // 4b. Fix content-after-object pattern (LLM places content string OUTSIDE the node object)
   270→   270→  // This is the SECOND MOST COMMON LLM error: content values appear as separate array elements
   271→   271→  // The LLM writes: {"id": "x", "style": {...}}, "Content Text"},
   272→   272→  // Should be:       {"id": "x", "style": {...}, "content": "Content Text"},
   273→   273→  //
   274→   274→  // The pattern is: double-close-brace, comma, string, then comma/close-bracket/parent-close
   275→   275→  // We need to: remove one close-brace, add "content" property, close the node, keep the separator
   276→   276→  // Note: (?:[^"\\]|\\.)* handles escaped quotes inside strings (e.g., "He said \"hello\"")
   277→   277→  
   278→   278→  // Fix: }}, "STRING"}, → }, "content": "STRING"},
   279→   279→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(,)/g, '}, "content": "$1"$2');
   280→   280→  // Fix: }}, "STRING"}] → }, "content": "STRING"}]
   281→   281→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\])/g, '}, "content": "$1"}$2');
   282→   282→  // Fix: }}, "STRING"} followed by another { → }, "content": "STRING"}, {
   283→   283→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\{)/g, '}, "content": "$1"}, $2');
   284→   284→  // Fix: }}, "STRING"}} → }, "content": "STRING"}}  (parent close after)
   285→   285→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\}\})/g, '}, "content": "$1"}$2');
   286→   286→  // Fix: }}, "STRING"}  (end of input or other)
   287→   287→  s = s.replace(/\}\},\s*"((?:[^"\\]|\\.){1,500})"\s*(\})/g, '}, "content": "$1"$2');
   288→   288→
   289→   289→  // 5. Fix key="value" patterns → "key": "value"
   290→   290→  // Handles: key="value", key = "value", key= "value", key ="value"
   291→   291→  s = s.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*=\s*"/g, '$1"$2": "');
   292→   292→
   293→   293→  // 6. Fix key='value' patterns → "key": "value"
   294→   294→  s = s.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*=\s*'/g, '$1"$2": "');
   295→   295→
   296→   296→  // 7. Fix unquoted property names (e.g., {display: "flex"} → {"display": "flex"})
   297→   297→  // Match: after { or , followed by an identifier followed by :
   298→   298→  // But be careful not to re-quote already-quoted keys
   299→   299→  s = s.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*:/g, (match, prefix, key) => {
   300→   300→    // Check if the key is already quoted (shouldn't match in that case, but just in case)
   301→   301→    if (prefix.endsWith('"') || prefix.endsWith("'")) return match;
   302→   302→    return `${prefix}"${key}":`;
   303→   303→  });
   304→   304→
   305→   305→  // 7. Fix single quotes used as string delimiters (not apostrophes inside strings)
   306→   306→  // Only replace single quotes that appear to be string delimiters:
   307→   307→  // - After : or [ or , followed by ' (opening quote)
   308→   308→  // - Before : or ] or , preceded by ' (closing quote)
   309→   309→  // Do NOT blindly replace all ' with " as it breaks strings like "I'll"
   310→   310→  // Since the LLM typically uses double quotes already, skip this step if no single-quote delimiters found
   311→   311→  if (/[:\[,]\s*'[^']*'\s*[:,\]}]/.test(s)) {
   312→   312→    // Replace opening single quotes after : [ ,
   313→   313→    s = s.replace(/([:\[,]\s*)'/g, '$1"');
   314→   314→    // Replace closing single quotes before : , ] }
   315→   315→    s = s.replace(/'(\s*[:,\]}])/g, '"$1');
   316→   316→  }
   317→   317→
   318→   318→  // 7c. Fix spaces in rgba/hsl function values (LLM often writes "rgba(0, 0, 0, 0.1)" instead of "rgba(0,0,0,0.1)")
   319→   319→  // This matters because the LLM sometimes outputs unquoted values like: "boxShadow": 0 4px 6px rgba(0, 0, 0, 0.1)
   320→   320→  // and the spaces in rgba() cause parse issues when we try to quote the value later
   321→   321→  s = s.replace(/rgba\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)\s*,\s*([\d.]+)\s*\)/g, 'rgba($1,$2,$3,$4)');
   322→   322→  s = s.replace(/hsl\(\s*(\d+)\s*,\s*(\d+)%\s*,\s*(\d+)%\s*\)/g, 'hsl($1,$2%,$3%)');
   323→   323→  s = s.replace(/hsla\(\s*(\d+)\s*,\s*(\d+)%\s*,\s*(\d+)%\s*,\s*([\d.]+)\s*\)/g, 'hsla($1,$2%,$3%,$4)');
   324→   324→
   325→   325→  // 7d. Fix unquoted CSS-like values after known CSS property names
   326→   326→  // The LLM sometimes writes: "boxShadow": 0 4px 6px rgba(0,0,0,0.1) (missing quotes around value)
   327→   327→  // or: "transform": rotate(45deg) (missing quotes around value)
   328→   328→  // Pattern: after a quoted CSS property name and colon, if the value starts with a digit, "rgba(", "hsl", "rotate", "translate", "scale", "linear-gradient", etc.
   329→   329→  // and is NOT already quoted, wrap it in quotes.
   330→   330→  // We look for: "cssProp": <unquoted-value> where unquoted-value continues until , } or ]
   331→   331→  const cssPropsWithUnquotedValues = [
   332→   332→    'boxShadow', 'textShadow', 'filter', 'backdropFilter', 'transform',
   333→   333→    'transition', 'animation', 'backgroundImage', 'background',
   334→   334→    'border', 'borderTop', 'borderRight', 'borderBottom', 'borderLeft',
   335→   335→    'outline', 'margin', 'padding', 'borderRadius',
   336→   336→  ];
   337→   337→  for (const prop of cssPropsWithUnquotedValues) {
   338→   338→    // Match: "prop": <value-not-starting-with-quote> up to comma, }, or ]
   339→   339→    // The value must NOT start with " or { or [ or true/false/null/number
   340→   340→    // We specifically target values that start with digits, letters, or rgba/hsl etc.
   341→   341→    const propRegex = new RegExp(
   342→   342→      `("${prop}")\\s*:\\s*(?!"|\\{|\\[|true|false|null)([a-zA-Z0-9().,\\s%#_-]+?)(\\s*[,}\\]])`,
   343→   343→      'g'
   344→   344→    );
   345→   345→    s = s.replace(propRegex, '$1: "$2"$3');
   346→   346→  }
   347→   347→
   348→   348→  // 8. Fix trailing commas before } or ]
   349→   349→  s = s.replace(/,\s*([}\]])/g, '$1');
   350→   350→
   351→   351→  // 9. Try to balance brackets/braces
   352→   352→  const openBraces = (s.match(/{/g) || []).length;
   353→   353→  const closeBraces = (s.match(/}/g) || []).length;
   354→   354→  const openBrackets = (s.match(/\[/g) || []).length;
   355→   355→  const closeBrackets = (s.match(/]/g) || []).length;
   356→   356→
   357→   357→  // Add missing closing brackets/braces at the end
   358→   358→  if (openBraces > closeBraces) {
   359→   359→    s += '}'.repeat(openBraces - closeBraces);
   360→   360→  }
   361→   361→  if (openBrackets > closeBrackets) {
   362→   362→    s += ']'.repeat(openBrackets - closeBrackets);
   363→   363→  }
   364→   364→
   365→   365→  // 10. Fix missing commas between properties (heuristic)
   366→   366→  // Pattern: "value" followed by whitespace then "key": — likely missing comma
   367→   367→  s = s.replace(/("\s*)(\n\s*"[a-zA-Z_]\w*"\s*:)/g, '$1,$2');
   368→   368→
   369→   369→  // 11. Validate by attempting to parse
   370→   370→  try {
   371→   371→    JSON.parse(s);
   372→   372→    return s;
   373→   373→  } catch {
   374→   374→    // If still invalid, try more aggressive fixes
   375→   375→    try {
   376→   376→      // Attempt progressive truncation from the end to find valid JSON
   377→   377→      for (let len = s.length; len > 10; len -= 1) {
   378→   378→        const truncated = s.substring(0, len);
   379→   379→        // Count brackets to ensure balance
   380→   380→        const ob = (truncated.match(/{/g) || []).length;
   381→   381→        const cb = (truncated.match(/}/g) || []).length;
   382→   382→        const obr = (truncated.match(/\[/g) || []).length;
   383→   383→        const cbr = (truncated.match(/]/g) || []).length;
   384→   384→        if (ob === cb && obr === cbr) {
   385→   385→          try {
   386→   386→            JSON.parse(truncated);
   387→   387→            return truncated;
   388→   388→          } catch {
   389→   389→            // continue
   390→   390→          }
   391→   391→        }
   392→   392→      }
   393→   393→    } catch {
   394→   394→      // Give up
   395→   395→    }
   396→   396→
   397→   397→    // 12. Last resort: try to extract just the design object by finding the deepest valid JSON prefix
   398→   398→    // This handles cases where the LLM generates a mostly-valid structure but has trailing garbage
   399→   399→    try {
   400→   400→      // Find the "design" key and try to extract its value
   401→   401→      const designKeyIdx = s.indexOf('"design"');
   402→   402→      if (designKeyIdx !== -1) {
   403→   403→        // Find the opening brace of the design value
   404→   404→        const designBraceIdx = s.indexOf('{', designKeyIdx + 8);
   405→   405→        if (designBraceIdx !== -1) {
   406→   406→          // Try to find a valid design object by scanning forward
   407→   407→          let depth = 0;
   408→   408→          let bestEnd = -1;
   409→   409→          for (let i = designBraceIdx; i < s.length; i++) {
   410→   410→            if (s[i] === '{') depth++;
   411→   411→            else if (s[i] === '}') depth--;
   412→   412→            if (depth === 0) {
   413→   413→              // We found a balanced design object, try to parse it
   414→   414→              const candidate = s.substring(designBraceIdx, i + 1);
   415→   415→              try {
   416→   416→                JSON.parse(candidate);
   417→   417→                bestEnd = i;
   418→   418→                // Don't break — keep looking for a larger valid object
   419→   419→              } catch {
   420→   420→                // This balanced section isn't valid JSON, continue
   421→   421→              }
   422→   422→            }
   423→   423→          }
   424→   424→          if (bestEnd !== -1) {
   425→   425→            const designObj = s.substring(designBraceIdx, bestEnd + 1);
   426→   426→            // Wrap in a response object
   427→   427→            const wrapped = `{"message": "Design recovered from partial response", "design": ${designObj}}`;
   428→   428→            try {
   429→   429→              JSON.parse(wrapped);
   430→   430→              return wrapped;
   431→   431→            } catch {
   432→   432→              // The design object alone might be valid
   433→   433→              try {
   434→   434→                JSON.parse(designObj);
   435→   435→                return designObj;
   436→   436→              } catch {
   437→   437→                // continue
   438→   438→              }
   439→   439→            }
   440→   440→          }
   441→   441→        }
   442→   442→      }
   443→   443→    } catch {
   444→   444→      // Give up
   445→   445→    }
   446→   446→
   447→   447→    return null;
   448→   448→  }
   449→   449→}

   451→   451→/**
   452→   452→ * Try to parse JSON with aggressive error repair.
   453→   453→ * Handles common LLM mistakes: unquoted keys, key="value", trailing commas,
   454→   454→ * single quotes, JS comments, extra text around JSON.
   455→   455→ */
   456→   456→function tryParseJSON(text: string): unknown | null {
   457→   457→  // First, try direct parse
   458→   458→  try {
   459→   459→    return JSON.parse(text);
   460→   460→  } catch {
   461→   461→    // continue to repair attempts
   462→   462→  }
   463→   463→
   464→   464→  // Use the aggressive repair function
   465→   465→  const repaired = repairLLMJson(text);
   466→   466→  if (repaired !== null) {
   467→   467→    try {
   468→   468→      return JSON.parse(repaired);
   469→   469→    } catch {
   470→   470→      // continue
   471→   471→    }
   472→   472→  }
   473→   473→
   474→   474→  // Additional targeted fixes if repairLLMJson failed
   475→   475→  let fixed = text;
   476→   476→
   477→   477→  // Fix 1: key="value" patterns (equals instead of colon)
   478→   478→  fixed = fixed.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*=\s*"/g, '$1"$2": "');
   479→   479→  try { return JSON.parse(fixed); } catch { /* continue */ }
   480→   480→
   481→   481→  // Fix 2: Unquoted property names
   482→   482→  fixed = fixed.replace(/([{,]\s*)([a-zA-Z_]\w*)\s*:/g, '$1"$2":');
   483→   483→  try { return JSON.parse(fixed); } catch { /* continue */ }
   484→   484→
   485→   485→  // Fix 3: Trailing commas
   486→   486→  fixed = fixed.replace(/,\s*([}\]])/g, '$1');
   487→   487→  try { return JSON.parse(fixed); } catch { /* continue */ }
   488→   488→
   489→   489→  // Fix 4: Single quotes to double quotes
   490→   490→  fixed = fixed.replace(/'/g, '"');
   491→   491→  try { return JSON.parse(fixed); } catch { /* continue */ }
   492→   492→
   493→   493→  // Fix 5: Remove JS comments
   494→   494→  fixed = fixed.replace(/\/\/.*$/gm, '');
   495→   495→  fixed = fixed.replace(/\/\*[\s\S]*?\*\//g, '');
   496→   496→  try { return JSON.parse(fixed); } catch { /* continue */ }
   497→   497→
   498→   498→  // Fix 6: Missing commas between properties
   499→   499→  fixed = fixed.replace(/("\s*)(\n\s*"[a-zA-Z_]\w*"\s*:)/g, '$1,$2');
   500→   500→  try { return JSON.parse(fixed); } catch { /* continue */ }
   501→   501→
   502→   502→  // Fix 7: Try balancing brackets
   503→   503→  const ob = (fixed.match(/{/g) || []).length;
   504→   504→  const cb = (fixed.match(/}/g) || []).length;
   505→   505→  const obr = (fixed.match(/\[/g) || []).length;
   506→   506→  const cbr = (fixed.match(/]/g) || []).length;
   507→   507→  if (ob > cb) fixed += '}'.repeat(ob - cb);
   508→   508→  if (obr > cbr) fixed += ']'.repeat(obr - cbr);
   509→   509→  try { return JSON.parse(fixed); } catch { /* continue */ }
   510→   510→
   511→   511→  return null;
   512→   512→}

   514→   514→export function parseAIResponse(response: string): {
   515→   515→  message: string;
   516→   516→  design: unknown | null;
   517→   517→  parseFailed?: boolean;
   518→   518→} {
   519→   519→  // Helper: extract design from parsed object
   520→   520→  function extractDesign(parsed: Record<string, unknown>): { message: string; design: unknown | null } | null {
   521→   521→    if (parsed.design || parsed.id || parsed.type || parsed.children) {
   522→   522→      return {
   523→   523→        message: (parsed.message as string) || 'Design generated successfully.',
   524→   524→        design: parsed.design || parsed,
   525→   525→      };
   526→   526→    }
   527→   527→    return null;
   528→   528→  }
   529→   529→
   530→   530→  // 1. Try to parse the full response as JSON
   531→   531→  const directParsed = tryParseJSON(response);
   532→   532→  if (directParsed && typeof directParsed === 'object') {
   533→   533→    const extracted = extractDesign(directParsed as Record<string, unknown>);
   534→   534→    if (extracted) return extracted;
   535→   535→  }
   536→   536→
   537→   537→  // 2. Try aggressive repair on the full response
   538→   538→  const repaired = repairLLMJson(response);
   539→   539→  if (repaired) {
   540→   540→    try {
   541→   541→      const repairedParsed = JSON.parse(repaired);
   542→   542→      if (repairedParsed && typeof repairedParsed === 'object') {
   543→   543→        const extracted = extractDesign(repairedParsed as Record<string, unknown>);
   544→   544→        if (extracted) return extracted;
   545→   545→      }
   546→   546→    } catch {
   547→   547→      // continue
   548→   548→    }
   549→   549→  }
   550→   550→
   551→   551→  // 3. Try to extract JSON from markdown code blocks (possibly multiple)
   552→   552→  const codeBlockRegex = /```(?:json)?\s*\n?([\s\S]*?)\n?```/g;
   553→   553→  let match;
   554→   554→  while ((match = codeBlockRegex.exec(response)) !== null) {
   555→   555→    const parsed = tryParseJSON(match[1]);
   556→   556→    if (parsed && typeof parsed === 'object') {
   557→   557→      const extracted = extractDesign(parsed as Record<string, unknown>);
   558→   558→      if (extracted) {
   559→   559→        // If there's text before the code block, use it as the message
   560→   560→        const beforeCode = response.substring(0, match.index).trim();
   561→   561→        if (beforeCode) {
   562→   562→          extracted.message = beforeCode.replace(/^[^\w]*/, '').trim() || extracted.message;
   563→   563→        }
   564→   564→        return extracted;
   565→   565→      }
   566→   566→    }
   567→   567→  }
   568→   568→
   569→   569→  // 4. Try to find the largest JSON object in the response
   570→   570→  const objectMatches = response.match(/\{[\s\S]*"id"[\s\S]*\}/g);
   571→   571→  if (objectMatches) {
   572→   572→    // Try from largest to smallest
   573→   573→    const sorted = objectMatches.sort((a, b) => b.length - a.length);
   574→   574→    for (const objMatch of sorted) {
   575→   575→      const parsed = tryParseJSON(objMatch);
   576→   576→      if (parsed && typeof parsed === 'object') {
   577→   577→        const extracted = extractDesign(parsed as Record<string, unknown>);
   578→   578→        if (extracted) return extracted;
   579→   579→      }
   580→   580→    }
   581→   581→  }
   582→   582→
   583→   583→  // 5. Try to find any JSON-like object with "id" (using unquoted key search too)
   584→   584→  const looseObjectMatches = response.match(/\{[\s\S]*(?:["']?id["']?\s*[:=])[\s\S]*\}/g);
   585→   585→  if (looseObjectMatches) {
   586→   586→    const sorted = looseObjectMatches.sort((a, b) => b.length - a.length);
   587→   587→    for (const objMatch of sorted) {
   588→   588→      const parsed = tryParseJSON(objMatch);
   589→   589→      if (parsed && typeof parsed === 'object') {
   590→   590→        const extracted = extractDesign(parsed as Record<string, unknown>);
   591→   591→        if (extracted) return extracted;
   592→   592→      }
   593→   593→    }
   594→   594→  }
   595→   595→
   596→   596→  // 6. Return as plain text message (strip markdown code blocks for cleaner output)
   597→   597→  const cleanedMessage = response
   598→   598→    .replace(/```(?:json)?\s*\n?/g, '')
   599→   599→    .replace(/\n?```/g, '')
   600→   600→    .trim();
   601→   601→  return { message: cleanedMessage || response, design: null, parseFailed: true };
   602→   602→}
